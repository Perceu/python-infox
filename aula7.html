Metodos especiais

					
					class Pessoa(object): 
						Nome = None 
						def __init__(self,nome):
							self.Nome = nome

						def andar(self, args): 
							print (self.Nome + ", Está andando ")

						def falar(self, args): 
							print (self.Nome + ", Está falando ")

						def __del__(self): 
							print (self.Nome + ", Estou morrendo")
